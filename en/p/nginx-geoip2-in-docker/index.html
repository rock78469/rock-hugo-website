# 1-1.Ôºöcompile GeoIP2 Package and Nginx
FROM debian:buster-slim AS builder
# FROM debian:buster-slim

# Install kits and tools
RUN apt-get update && \
    apt-get install -y libmaxminddb-dev wget build-essential && \
    rm -rf /var/lib/apt/lists/*

# Download„ÄÅcompile and install GeoIP2 module
RUN wget https://github.com/maxmind/libmaxminddb/releases/download/1.6.0/libmaxminddb-1.6.0.tar.gz && \
    tar xvfz libmaxminddb-1.6.0.tar.gz && \
    cd libmaxminddb-1.6.0 && \
    ./configure && \
    make && \
    make install && \
    ldconfig && \
    cd .. && \
    rm -rf libmaxminddb-1.6.0 libmaxminddb-1.6.0.tar.gz

# Download„ÄÅcompile and install ngx_http_geoip2_module module
RUN wget https://github.com/leev/ngx_http_geoip2_module/archive/master.tar.gz && \
    tar xvfz master.tar.gz && \
    rm -rf master.tar.gz && \
    mv ngx_http_geoip2_module-master /usr/src/nginx-geoip2

# Download„ÄÅunzip,and compile Pcre module
RUN wget http://downloads.sourceforge.net/project/pcre/pcre/8.35/pcre-8.35.tar.gz && \
    tar zxvf pcre-8.35.tar.gz && \
    cd pcre-8.35 && \
    ./configure && \
    make && make install && \
    cd ..
# Download„ÄÅunzip,and compile Zlib module
RUN wget http://zlib.net/zlib-1.2.13.tar.gz && \
    tar zxvf zlib-1.2.13.tar.gz && \
    cd zlib-1.2.13 && \
    ./configure && \
    make && make install && \
    cd ..

# Download„ÄÅunzip,and compile Nginx
RUN wget http://nginx.org/download/nginx-1.21.3.tar.gz && \
    tar xvfz nginx-1.21.3.tar.gz && \
    cd nginx-1.21.3 && \
    ./configure --with-compat --add-dynamic-module=/usr/src/nginx-geoip2 && \
    make modules && \
    cd ..

# 1-2ÔºöBuild final image
FROM nginx:1.21.3

# Copy ngx_http_geoip2_module module from compiled step1 
COPY --from=builder /nginx-1.21.3/objs/ngx_http_geoip2_module.so /etc/nginx/modules/

# Install GeoIP2 repo and tools
RUN apt-get update && \
    apt-get install -y libmaxminddb0 mmdb-bin && \
    rm -rf /var/lib/apt/lists/*

# Copy GeoIP2 repo
COPY --from=builder /usr/local/lib/libmaxminddb.so.* /usr/local/lib/

ADD ./GeoLite/GeoLite2-Country_20230221/GeoLite2-Country.mmdb /usr/share/GeoIP/

# Config Nginx to enable GeoIP2 module
# RUN sed -i '/http {/a \    geoip2 /usr/share/GeoIP/GeoLite2-Country.mmdb  {\n        auto_reload 60m;\n        $geoip2_metadata_country_build metadata build_epoch;\n        $geoip2_data_country_code country iso_code;\n    }' /etc/nginx/nginx.conf && \
#     sed -i '1 i load_module modules/ngx_http_geoip2_module.so;' /etc/nginx/nginx.conf

CMD ["nginx", "-g", "daemon off;"]

Step4. Apply GeoLite Database

Link

  2. After entering personal information, click Continue to complete the registration
  3. Download Download Page
  4. Select download GeoLite2 City
  5. Put the decompressed GeoLite2-City.mmdb into the GeoLite folder

version: '3.5'

services:
  nginx:
    image: mynginx:latest
    container_name: mynginx
    ports:
      - 8081:80
    volumes:
      - ./nginx/config/nginx.conf:/etc/nginx/nginx.conf
      - ./nginx/log:/var/log/nginx
      - ./GeoLite/GeoLite2-City.mmdb:/usr/share/GeoIP
    networks:
      - nginx_network

networks:
  nginx_network
    name: nginx_network
    driver: bridge
