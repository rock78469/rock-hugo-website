apiVersion: apps/v1
kind: Deployment
metadata:
  name: imageproxy-deployment
  labels:
    app: imageproxy
spec:
  replicas: 2
  selector:
    matchLabels:
      app: imageproxy
  template:
    metadata:
      labels:
        app: imageproxy
    spec:
      containers:
        - name: imageproxy
          image: "willnorris/imageproxy"
          resources:
            requests:
              memory: 50Mi
              cpu: 50m
            limits:
              memory: 100Mi
              cpu: 50m
          env:
            - name: IMAGEPROXY_ADDR
              value: "0.0.0.0:8080"
            - name: IMAGEPROXY_ALLOWHOSTS
              value: "storage.googleapis.com"
          ports:
          - containerPort: 8080
            name: http-imageproxy
          volumeMounts:
            - mountPath: /cache
              name: cache-volume
      volumes:
        - name: cache-volume
          emptyDir: {}
---
apiVersion: v1
kind: Service
metadata:
  name: service-imageproxy
  labels:
    service: service-imageProxy
spec:
  type: ClusterIP
  selector:
    app: imageproxy
  ports:
    - protocol: TCP
      port: 8080
      name: tcp-imageproxy
      targetPort: 8080
