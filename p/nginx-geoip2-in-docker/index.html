# 第一階段：編譯GeoIP2庫和Nginx模塊
FROM debian:buster-slim AS builder
# FROM debian:buster-slim

# 安裝必要的套件和工具
RUN apt-get update && \
    apt-get install -y libmaxminddb-dev wget build-essential && \
    rm -rf /var/lib/apt/lists/*

# 下載、編譯和安裝GeoIP2庫
RUN wget https://github.com/maxmind/libmaxminddb/releases/download/1.6.0/libmaxminddb-1.6.0.tar.gz && \
    tar xvfz libmaxminddb-1.6.0.tar.gz && \
    cd libmaxminddb-1.6.0 && \
    ./configure && \
    make && \
    make install && \
    ldconfig && \
    cd .. && \
    rm -rf libmaxminddb-1.6.0 libmaxminddb-1.6.0.tar.gz

# 下載、編譯和安裝ngx_http_geoip2_module模塊
RUN wget https://github.com/leev/ngx_http_geoip2_module/archive/master.tar.gz && \
    tar xvfz master.tar.gz && \
    rm -rf master.tar.gz && \
    mv ngx_http_geoip2_module-master /usr/src/nginx-geoip2

# 下載、解壓縮和編譯Pcre
RUN wget http://downloads.sourceforge.net/project/pcre/pcre/8.35/pcre-8.35.tar.gz && \
    tar zxvf pcre-8.35.tar.gz && \
    cd pcre-8.35 && \
    ./configure && \
    make && make install && \
    cd ..
# 下載、解壓縮和編譯Zilb
RUN wget http://zlib.net/zlib-1.2.13.tar.gz && \
    tar zxvf zlib-1.2.13.tar.gz && \
    cd zlib-1.2.13 && \
    ./configure && \
    make && make install && \
    cd ..

# 下載、解壓縮和編譯Nginx
RUN wget http://nginx.org/download/nginx-1.21.3.tar.gz && \
    tar xvfz nginx-1.21.3.tar.gz && \
    cd nginx-1.21.3 && \
    ./configure --with-compat --add-dynamic-module=/usr/src/nginx-geoip2 && \
    make modules && \
    cd ..

# 第二階段：構建最終映像
FROM nginx:1.21.3

# 複製第一階段編譯的ngx_http_geoip2_module模塊
COPY --from=builder /nginx-1.21.3/objs/ngx_http_geoip2_module.so /etc/nginx/modules/

# 安裝GeoIP2庫和相關工具
RUN apt-get update && \
    apt-get install -y libmaxminddb0 mmdb-bin && \
    rm -rf /var/lib/apt/lists/*

# 複製GeoIP2庫
COPY --from=builder /usr/local/lib/libmaxminddb.so.* /usr/local/lib/

ADD ./GeoLite/GeoLite2-Country_20230221/GeoLite2-Country.mmdb /usr/share/GeoIP/

# 配置Nginx，啟用GeoIP2模塊
# RUN sed -i '/http {/a \    geoip2 /usr/share/GeoIP/GeoLite2-Country.mmdb  {\n        auto_reload 60m;\n        $geoip2_metadata_country_build metadata build_epoch;\n        $geoip2_data_country_code country iso_code;\n    }' /etc/nginx/nginx.conf && \
#     sed -i '1 i load_module modules/ngx_http_geoip2_module.so;' /etc/nginx/nginx.conf

CMD ["nginx", "-g", "daemon off;"]

load_module modules/ngx_http_geoip2_module.so;

user  nginx;
worker_processes  auto;

error_log  /var/log/nginx/error.log notice;
pid        /var/run/nginx.pid;


events {
    worker_connections  1024;
}


http {
    geoip2 /usr/share/GeoIP/GeoLite2-Country.mmdb  {
        auto_reload 60m; # 自動重新載入時間間隔
        $geoip2_data_city_name city names en; # 市名
        $geoip2_data_city_latitude location latitude; # 緯度
        $geoip2_data_city_longitude location longitude; # 經度
        $geoip2_data_country_iso_code country iso_code; # 國家ISO代碼
    }
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;

    # log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
    #                   '$status $body_bytes_sent "$http_referer" '
    #                   '"$http_user_agent" "$http_x_forwarded_for"';
    # log_format custom '$remote_addr - $remote_user [$time_local]'
    #                   '"$request" $status $body_bytes_sent'
    #                   '"$http_referer" $host "$http_user_agent"'
    #                   '"$request_time" "$upstream_connect_time"'
    #                   '"$geoip2_data_city_name" "$geoip2_data_country_iso_code"';

    log_format main2 '{"remote_addr":"$remote_addr","remote_user":"$remote_user","time":"$time_local","request":"$request","status":"$status","body_bytes_sent":"$body_bytes_sent","http_referer":"$http_referer","host":"$host","http_user_agent":"$http_user_agent","request_time":"$request_time","upstream_connect_time":"$upstream_connect_time","geoip2_data_city_name":"$geoip2_data_city_name","geoip2_data_country_iso_code":"$geoip2_data_country_iso_code","geoip2_data_city_latitude":"$geoip2_data_city_latitude","geoip2_data_city_longitude":"$geoip2_data_city_longitude"}';

    access_log  /var/log/nginx/access.log  main2;

    sendfile        on;
    #tcp_nopush     on;

    keepalive_timeout  65;

    #gzip  on;

    include /etc/nginx/conf.d/*.conf;
}
}

Step4. 申請GeoLite 資料庫

註冊連結

  1. 點選註冊GeoIP2
  2. 輸入個人資料後點選 Continue，註冊完成
  3. 點選下載 下載頁面
  4. 下載GeoLite2 City
  5. 將解壓縮後的 GeoLite2-City.mmdb 放到GeoLite資料夾

version: '3.5'

services:
  nginx:
    image: mynginx:latest
    container_name: mynginx
    ports:
      - 8081:80
    volumes:
      - ./nginx/config/nginx.conf:/etc/nginx/nginx.conf
      - ./nginx/log:/var/log/nginx
      - ./GeoLite/GeoLite2-City.mmdb:/usr/share/GeoIP
    networks:
      - nginx_network

networks:
  nginx_network
    name: nginx_network
    driver: bridge
